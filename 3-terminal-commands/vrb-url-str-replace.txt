VRB database str replace
This replaces url structure from production to local and visa versa

TO LOCAL
php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s www.vanbreda.be -r vrb.loc && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s www.vanbreda.com -r vrb.loc/international && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s www.artsecure.be -r vrb.loc/artsecure && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s www.jobsatvanbreda.be -r vrb.loc/jobsatvanbreda && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s www.justitia.be -r vrb.loc/justitia

TO PRODUCTION
php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s staging.vanbreda.be -r vrb.loc && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s staging.vanbreda.be/international -r vrb.loc/international && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s staging.vanbreda.be/artsecure -r vrb.loc/artsecure && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s staging.vanbreda.be/jobsatvanbreda -r vrb.loc/jobsatvanbreda && php ~/Desktop/sr/srdb.cli.php -h localhost -n vrb -u root -p root -s staging.vanbreda.be/justitia -r vrb.loc/justitia
